#include <stdio.h>
 #include <stdlib.h>

 typedef struct element {
   struct element *next;
   unsigned long val;
 } el_listy;

 void dodaj_do_listy (el_listy *lista, unsigned long liczba)//zmiana11
 {//zmiana12
   el_listy *wsk, *nowy;//zmiana13
   wsk = lista;//zmiana14
   while (wsk->next != NULL)//zmiana15
     { //zmiana16
     wsk = wsk->next; /* przesuwamy wsk a¿ znajdziemy ostatni element *//zmiana17
     }//zmiana18
   nowy =(el_listy*) malloc (sizeof(el_listy));//zmiana19
   nowy->val = liczba;//zmiana200
   nowy->next = NULL;
   wsk->next = nowy; /* podczepiamy nowy element do ostatniego z listy */
 }
 
 void wypisz_liste(el_listy *lista)
 {
   el_listy *wsk=lista;
   while( wsk != NULL )
     {
     printf ("%lu\n", wsk->val);
     wsk = wsk->next;
     }
 }
 
 int jest_pierwsza(el_listy *lista, int liczba)
 {
   el_listy *wsk;
   wsk = lista;
   while (wsk != NULL) {
     if ((liczba%wsk->val)==0) return 0;
        wsk = wsk->next;
     }
     return 1;
 }
 
 int main ()
 {
   unsigned long i = 3; /* szukamy liczb pierwszych w zakresie od 3 do 1000 */
   const unsigned long END = 1000;
   first =(el_listy*) malloc (sizeof(el_listy));
   first->val = 2;

   for (;i!=END;++i) {
